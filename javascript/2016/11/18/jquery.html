sortTable

//  sortTable(f,n)
//  f : 1 ascending order, -1 descending order
//  n : n-th child(<td>) of <tr>

function sortTable(f,n){
  var rows = $('#entities-table tbody  tr').get();
  rows.sort(function(a, b) {
    var A = getVal(a);
    var B = getVal(b);

    if(A < B) {
        return -1*f;
    }
    if(A > B) {
      return 1*f;
    }
    return 0;
  });

  function getVal(elm){
    var v = $(elm).children('td').eq(n).text().toUpperCase();
    if(!v && $(elm).children('td').eq(n).find('input').length) {
      v = $(elm).children('td').eq(n).children('input').is(':checked');
    }
    if($.isNumeric(v)){
      v = parseInt(v,10);
    }
    //console.log(v);
    return v;
  }

  $.each(rows, function(index, row) {
    $('#entities-table').children('tbody').append(row);
  });
}

jshint

jslint used to be the preferred tool for static JavaScript checking. Later it was forked into jshint.

jshint foo.js

The preferred way to configure jshint is to create a json configure file called .jshintrc.

jshint -c .jshintrc foo.js

You can see current list of all available options here and their explanations.
