class MyHandler extends Handler {
    public void handleMessage(Message msg) {
        Bundle bundle = msg.getData();
        Date date = new Date();
        textView.append((new SimpleDateFormat("hh:mm:ss")).format(date) +
                        "    "+ bundle.getString("msg") + "\n");
    }
}
