## run

```
function run() {
    var image = document.getElementById('background');
    image.onload = function() {
        var engine = new RainyDay({
            image: this
        });
        engine.rain([ [0, 2, 200], [3, 3, 1] ], 100);
    };
    image.crossOrigin = 'anonymous';
    image.src='example.jpg';
}
```
