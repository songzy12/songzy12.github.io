## sortTable

```
//  sortTable(f,n)
//  f : 1 ascending order, -1 descending order
//  n : n-th child(<td>) of <tr>

function sortTable(f,n){
  var rows = $('#entities-table tbody  tr').get();
  rows.sort(function(a, b) {
    var A = getVal(a);
    var B = getVal(b);

    if(A < B) {
        return -1*f;
    }
    if(A > B) {
      return 1*f;
    }
    return 0;
  });

  function getVal(elm){
    var v = $(elm).children('td').eq(n).text().toUpperCase();
    if(!v && $(elm).children('td').eq(n).find('input').length) {
      v = $(elm).children('td').eq(n).children('input').is(':checked');
    }
    if($.isNumeric(v)){
      v = parseInt(v,10);
    }
    //console.log(v);
    return v;
  }

  $.each(rows, function(index, row) {
    $('#entities-table').children('tbody').append(row);
  });
}
```

## getJSON

`$.getJSON(url, data, func)` sends a GET request to url and will send the contents of the data object as query parameters. Once the data arrived, it will call the given function with the return value as argument.

```
$(document).ready(function() {
    function video_time_update() {
        var videoplayer = $('video').get(0);
```

## jshint

`jslint` used to be the preferred tool for static JavaScript checking. Later it was forked into `jshint`.

```
jshint foo.js
```

The preferred way to configure jshint is to create a json configure file called `.jshintrc`.

```
jshint -c .jshintrc foo.js
```

You can see current list of all available options [here](https://github.com/jshint/jshint/blob/master/examples/.jshintrc) and their explanations.

```
{
    "esnext": true, // Allow ES.next (ES6) syntax (ex: `const`) 
    "indent": 2,
    "maxlen": 80, // Max number of characters per line
    "freeze": true, // prohibits overwriting prototypes of native objects such as Array, Date etc.
    "camelcase": true, // Identifiers must be in camelCase
    "unused": true, // all variables, last function parameter
    "eqnull": true, // Tolerate use of `== null`
    "proto": true, // Tolerate using the `__proto__` property
    "supernew": true, // Tolerate `new function () { ... };` and `new Object;`
    "noyield": true, // Tolerate generator functions with no yield statement in them.
    "evil": true, // Tolerate use of `eval` and `new Function()
    "node": true, // Node.js
    "boss": true, // Tolerate assignments where comparisons would be expected
    "expr": true, // Tolerate `ExpressionStatement` as Programs
    "loopfunc": true, // Tolerate functions being defined in loops
    "white": true, // Check against strict whitespace and indentation rules
    "maxdepth": 4 // Max depth of nested blocks (within functions)
}
```
