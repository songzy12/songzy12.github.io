```
sqlite3 user
create table tbl(name char(10), age smallint, score float);
.table
insert into tbl values('yanggang', 24, 98);
insert into tbl values('sunboy', 20, 78.5);
select * from tbl;
.mode column
.databases
.backup main user.sql
.output user_tbl.sql
.dump tbl
```
