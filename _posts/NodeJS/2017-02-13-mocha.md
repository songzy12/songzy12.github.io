## test.js

```
const request = require('supertest')
const assert = require('assert')
const api = request(require('../app'));

const adminUser = {
  username: 'user',
  password: '123456',
  token: null,
}

function graphQl(query, variables) {
  var variables = variables || {};
  return api.post('/api').set('content-type', 'application/json').send(JSON.stringify({
    query,
    variables
  }));
}

describe('admin', function () {
  it('login', function (done) {
    graphQl('mutation($username: String!, $password: String!)' +
      ' {createToken(username: $username, password: $password) {apiKey, isSuperAdmin}}',
      JSON.stringify(adminUser)
    ).expect(200).end(function (err, rsp) {
      adminUser.token = rsp.body.data.createToken.apiKey;
      done();
    });
  });
});

describe('user-api', function () {
  const newUserNTID = 'id-' + new Date().getTime();
  it('get all', function (done) {
    graphQl(`query($page: Int!, $criteria: String) { 
      users(page: $page, criteria: $criteria) 
        { docs {ntId, hasRetail360Access, isAdmin, firstName, lastName} 
          total, limit, page, pages
        } 
      }`, {page: 1, criteria: ''}).expect(200).end(function (err, rsp) {
      done();
    });
  })

  it('create', function (done) {
    graphQl('mutation($token: String!, $user: UserInputCreate!) { addUser(token: $token, user: $user) { ntId } }',
      {
        user: {
          firstName: 'fname-unit-test',
          lastName: 'lname-unit-test',
          ntId: newUserNTID,
          hasRetail360Access: true,
          isAdmin: true
        },
        token: adminUser.token
      }).expect(200).end(function (err, rsp) {
      assert(rsp.body.data.addUser.ntId === newUserNTID);
      done();
    })
  });

  it('update', function (done) {
    graphQl(
      'mutation($ntId: String!, $user: UserInputUpdate!, $token: String!) ' +
      '{ updateUser(token: $token, ntId: $ntId, user: $user) { ntId } }',
      {
        ntId: newUserNTID,
        user: {
          firstName: 'fname-unit-test-updated',
          lastName: 'lname-unit-test',
          hasRetail360Access: false,
          isAdmin: false
        },
        token: adminUser.token
      })
      .expect(200, done);
  });

  it('find', function (done) {
    graphQl(`query($page: Int!, $criteria: String) {
      users(page: $page, criteria: $criteria)
        { docs {ntId, hasRetail360Access, isAdmin, firstName, lastName}
          total, limit, page, pages
        }
      }`, {page: 1, criteria: newUserNTID}).expect(200).end(function (err, rsp) {
      assert(rsp.body.data.users.docs[0].ntId === newUserNTID);
      done();
    });
  })

  it('delete', function (done) {
    graphQl(
      'mutation($ntId: String!, $token: String!) { deleteUser(token: $token, ntId: $ntId) { ntId } }', {
        ntId: newUserNTID,
        token: adminUser.token
      }).expect(200, done);
  });
});
```
