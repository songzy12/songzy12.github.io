const graphql = require('graphql');
const model = require('mongoose').model(modelName);
const {omit, assign} = require('lodash/fp');

  // All the query fields
  queryFields = {

    user: {
      type,
      args: { ntId },
      resolve: (_, {ntId}) => model.findOne({ntId})
    },
    users: {
      type: new graphql.GraphQLObjectType({
        name: modelName + 's',
        fields: assign({ docs: { type: new graphql.GraphQLList(type) }}, paginationFields)
      }),

      args: assign({
        criteria: {type: graphql.GraphQLString},
        sortName: {type: graphql.GraphQLString},
        sortOrder: {type: graphql.GraphQLString}
      }, omit('total, pages', paginationFields)),
      resolve: (_, args) => {

        let query = {};
        if (args.criteria) {
          query = {
            '$or': [
              {'ntId': {'$regex': args.criteria, '$options': 'i'}},
              {'firstName': {'$regex': args.criteria, '$options': 'i'}},
              {'lastName': {'$regex': args.criteria, '$options': 'i'}}]
          };
        }
        if (args.sortName && args.sortOrder) {
          args['sort'] = {}
          args['sort'][args.sortName] = args.sortOrder === 'asc' ? 1 : -1;
        }
        return model.paginate(query, omit('token, criteria', args))
      }
    },
  },
  // All the mutation fields
  mutationFields = {
    addUser: {
      type,
      args: {token, user: {type: inputTypeCreate}},
      resolve: (_, {user}) => new model(user).save()
    },
    updateUser: {
      type,
      args: {token, ntId, user: {type: inputTypeUpdate}},
      resolve: (_, {ntId, user}) => model.findOneAndUpdate({ntId}, user, {new: true, upsert: true})
    },
    deleteUser: {
      type,
      args: {token, ntId},
      resolve: (_, {ntId}) => model.findOneAndRemove({ntId})
    }
  };

module.exports = {
  queryFields,

## paginate

/**
 * Generates the pagination below the table
 * @param pages The total pages
 */
function generatePagination(pages) {
  let pagination;
  if(pages < 2) pagination = '';
  else {
    const allPages = Array.apply(null, {length: pages})
      .map((c, i) => `<li><a href="javascript:;" id="page_index${i + 1}" class="page-index" onclick="paginate(${i + 1})">${i + 1}</a></li> `);
    pagination = `
    <ul class="pagination">
      <li>
        <a href="javascript:;" aria-label="Previous" onclick="prev()">
          <span aria-hidden="true">&laquo;</span>
        </a>
      </li>
      ${allPages.join(' ')}
      <li>
        <a href="javascript:;" aria-label="Next" onclick="next()">
          <span aria-hidden="true">&raquo;</span>
        </a>
      </li>
    </ul>
 `;
  }
  $('#pagination').html(pagination);
}

/**
 * Go to the given page number
 * @param page The page number
 */
function paginate(page) {
  currentPage = page;
  refreshEntities();
}

/**

function buildTable(element, opts) {
  const tableElement = $(element);
  opts = opts || {};

  tableElement.html("");
  const tHead = $(`<thead></thead>`).appendTo(tableElement);
  const hRow = $('<tr></tr>').appendTo(tHead);
  const tableSortPrefix= 'table-sort-';
  $.each(opts.headers, (i, h) => {
    const tmpSortName = opts.sortFields[i];
    hRow.append(`<th id="${tableSortPrefix + tmpSortName}" class="th-witdh-${opts.headersWidth[i]}"> ${h} 
                    <span class="sort-arrow-hide sort-arrow glyphicon glyphicon-triangle-top"></span>
                 </th>`);
    //click event
    $('#' + tableSortPrefix + tmpSortName).click(function () {
      const name = $(this).attr('id').substr(tableSortPrefix.length);

      if (name !== sortName) {
        const arrow = $('#' + tableSortPrefix + sortName).find('.sort-arrow');
        arrow.removeClass('glyphicon-triangle-top glyphicon-triangle-bottom');
        arrow.addClass('sort-arrow-hide glyphicon-triangle-top');
      }

      const sortArrow = $(this).find('.sort-arrow');
      if (sortArrow.hasClass('sort-arrow-hide')) {
        sort(name, 'asc');
      } else {

          sort(name, 'desc');
        } else {
          sort(name, 'asc');
        }
      }
    });

    if (sortName.length > 0 && sortOrder.length > 0) {
