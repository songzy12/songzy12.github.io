/**
 * Runs the graphql query for import
 * @param schema The graphql schema
 * @param query The graphql query
 * @param variables The variables
 * @param callback The callback to call after running the query
 */
function runGraphQl(schema, query, variables, callback) {
  graphql(schema, query, {},{}, variables)
    .then(function (result) {
      let r = {};
      if (!result.errors) r = {ret: 'success', data: result.data}
      else r = {ret: 'failed', data: result.errors}
      r = JSON.stringify(r) + '|||'
      callback(null, r);
    })
    .catch(callback);
}

function showImportErrors(totalNumber,successedNumber,faileds , modelName) {

     but the following ${failed.length} number of users failed.`
    : `All ${modelName.toLowerCase()}s import failed.`;

  let content = '<b>' + title + '</b>';
  content += '<ul>';
  for (let i = 0; i < faileds.length; i++) {
    content += `<li> ${modelName} failed to save due to ${faileds[i].data[0].message}</li>`;
  }
  content += '</ul>';
  showAlert(content, totalNumber > faileds.length ? 'alert-success' : 'alert-danger');
}

/**
 * Attach jquery file uploader to the element
 * @param token The auth token
 * @param refreshFn The refresh function
 * @param modelName , the model name
 */
function attachFileUploader(token, refreshFn, modelName) {
  $('#fileupload').fileupload({
    sequentialUploads: true,
    acceptFileTypes: /\.(csv)$/i,
    beforeSend: function(xhr) {
      xhr.setRequestHeader('X-AuthToken', token.apiKey);
    },
    done: function (e, data) {
      const resultsStrArray = data.result.split('|||');
      let successedNumber = 0;
      const faileds = [];
      for (let i = 0; i < resultsStrArray.length - 1; i++) {
        const r = JSON.parse(resultsStrArray[i]);
        if (r.ret === 'success') {
          successedNumber += 1;
        } else {
          faileds.push(r);
        }
      }
      if (faileds.length == 0) {
        showAlert(`${successedNumber} number of ${modelName.toLowerCase()}s successfully saved.`, 'alert-success');
      } else {
        showImportErrors(resultsStrArray.length - 1, successedNumber, faileds, modelName);
      }
      refreshFn();
    },
    fail: function(event, err) {
      let rspTxt = err.jqXHR.responseText;
      try{
        const errorJson = JSON.parse(rspTxt);
        rspTxt = errorJson[0].message || rspTxt;
      }catch (e){}
      showAlert('Import Failed due to ' + (rspTxt || 'server inner error') + '.' , 'alert-danger');
    }
  });
}
