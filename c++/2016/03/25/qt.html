#include "mainwindow.h"
#include <QApplication>

QApplication app(argc, argv);
MainWindow w;
w.show();
int ret = app.exec();

MainWindow::MainWindow(QWidget *parent) :
    QMainWindow(parent),
    ui(new Ui::MainWindow)
{
    ui->setupUi(this);
    connect(ui->pushButton, SIGNAL(pressed()), this, SLOT(startSynth()));
}

void MainWindow::mousePressEvent(QMouseEvent *event){
    if (event->buttons() & Qt::LeftButton){
        int mouse_x = QCursor::pos().x();
        int mouse_y = QCursor::pos().y();
        QWidget *action = QApplication::widgetAt(mouse_x, mouse_y);
        QPoint GlobalPoint(action->mapToGlobal(QPoint(0, 0)));
        int x = GlobalPoint.x();
        int y = GlobalPoint.y();
        ui->horizontalSlider_1->setValue(x);
        ui->horizontalSlider_4->setValue(y);
    }
}
