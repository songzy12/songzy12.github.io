test.js

process.env.NODE_ENV = 'test';

let chai = require('chai');
let chaiHttp = require('chai-http');
let server = require('../../app');
let common = require('../common')

let User = require("mongoose").model('User');
let Admin = require("mongoose").model('Admin');

let should = chai.should();

chai.use(chaiHttp);

describe('Hompage', () => {
  it('/ should redirect to /admin/login.html when not logged in', () => {
    browser.url('http://localhost:3000');
    var title = browser.getUrl();
    title.should.equal('http://localhost:3000/admin/login.html');
  });
});

describe('Login and Users page', () => {
  before(() => {
    common.clearModel(Admin)
      .then(() => common.createAdmin(Admin))
      .then(() => {
        return common.makeRequest(
          chai,
          server,
          'mutation($username: String!, $password: String!) {createToken(username: $username, password: $password) {apiKey, isSuperAdmin}}',
          {username: 'admin', password: 'admin'}
        );
      }).then((result) => {
        const token = result.body.data.createToken.apiKey;
        return common.uploadUsers(chai, server, 'test-data/users.csv', token);
      });
  });

  after(() => {
    common.clearModel(Admin)
    .then(() => common.clearModel(User));
  })

  it('should login correctly', () => {
    const tableSelector = '#entities-table > tbody > tr';
    browser.url('http://localhost:3000/admin/')
      .setValue('#inputEmail', 'admin')
      .setValue('#inputPassword', 'admin')
      .click('#sign-in')
      .waitUntil(() => browser.elements(tableSelector).value.length > 0);
    const rows = browser.elements(tableSelector);
    rows.value.length.should.equal(4);
  });
});
